var buildTree = function(preorder, inorder) { if(preorder.length == 1){ return new TreeNode(preorder[0]) //返回的不该是数，而是一个结点 } if(inorder.length == 0){ return null } if(inorder.length > 1){ var rootVal = preorder.shift() var idx = inorder.indexOf(rootVal) var inL = inorder.slice(0,idx) var inR = inorder.slice(idx +1) var pL = preorder.slice(0,inL.length) var pR = preorder.slice(inL.length) var root = new TreeNode(rootVal) root.left = buildTree(pL,inL) root.right = buildTree(pR,inR) return root } };
